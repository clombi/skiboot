      36                 :            : /// \brief Inline PGAS assembler for PgP/Stage1 PORE
      37                 :            : ///
      38                 :            : /// \page pore_inline_assembler PORE Inline Assembler and Disassembler
      39                 :            : ///
      40                 :            : /// Several procedures targeting the PORE engine require inline assembly and
      41                 :            : /// disassembly of PORE code, that is, they require that PORE instructions be
      42                 :            : /// assembled/disassembled directly into/from a host memory buffer. This page
      43                 :            : /// describes these facilities.  The APIs described here are implemented in
      44                 :            : /// the files pore_inline.h, pore_inline_assembler.c and
      45                 :            : /// pore_inline_disassembler.c.  Both the inline assembelr and disassembler
      46                 :            : /// conform to the PGAS assembly format for PORE.
      47                 :            : ///
      48                 :            : /// Both inline assembly and disassembly make use of a PoreInlineContext
      49                 :            : /// structure. This structure represents the state of a memory area being
      50                 :            : /// targeted for inline assembly and disassembly.  The context is initialized
      51                 :            : /// with the pore_inline_context_create() API, and a pointer to an instance of
      52                 :            : /// this structure appears as the first argument of all assembler/disassembler
      53                 :            : /// APIs. As assembly/disassembly progresses the PoreInlineContext keeps
      54                 :            : /// track of how much host memory area has been filled by assembled code or
      55                 :            : /// scanned by the disassebler.
      56                 :            : ///
      57                 :            : /// Assembler/disassembler APIs are predicates that return 0 for success and a
      58                 :            : /// non-zero error code for failure.  In the event of failure, the error code
      59                 :            : /// (a small integer) is also stored in the \a error field of the context
      60                 :            : /// structure.  String forms of the error codes are also available in the
      61                 :            : /// global array pore_inline_error_strings[].
      62                 :            : ///
      63                 :            : /// The assembler always produces PORE code in the PORE-native big-endian
      64                 :            : /// format.  Likewise, the diassembler assumes the host memory to be
      65                 :            : /// disassembled contains PORE code in big-endian format.
      66                 :            : ///
      67                 :            : /// \section Initialization
      68                 :            : ///
      69                 :            : /// Before invoking inline assembly/disassembly APIs, an instance of a
      70                 :            : /// PoreInlineContext structure must be initialized using the
      71                 :            : /// pore_inline_context_create() API.  For assembly, the context describes the
      72                 :            : /// host memory buffer that will contain the assembled code.  For disassembly,
      73                 :            : /// the context describes the host memory area that contains the code to be
      74                 :            : /// disassembled. Full documentation is available for
      75                 :            : /// pore_inline_context_create(), including documentation for options that
      76                 :            : /// control assembly and disassembly.  The implementation also provides a
      77                 :            : /// 'copy operator' for the context, pore_inline_context_copy().
      78                 :            : ///
      79                 :            : /// An example of initializing a context for inline assembly with parity
      80                 :            : /// checking appears below.
      81                 :            : ///
      82                 :            : /// \code 
      83                 :            : ///
      84                 :            : /// PoreInlineContext ctx;
      85                 :            : /// uint32_t buf[BUFSIZE];
      86                 :            : ///
      87                 :            : /// rc = pore_inline_context_create(&ctx, buf, BUFSIZE * 4, 0,
      88                 :            : ///                                 PORE_INLINE_CHECK_PARITY);
      89                 :            : /// if (rc) . . . Handle Error
      90                 :            : ///
      91                 :            : /// \endcode
      92                 :            : ///
      93                 :            : /// Applications that reuse the same memory buffer for assembling and
      94                 :            : /// processing multiple PORE programs can 'reset' the context between uses by
      95                 :            : /// using the pore_inline_context_reset() API.  pore_inline_context_reset()
      96                 :            : /// resets the location counter and memory extent to their initial (creation)
      97                 :            : /// values, and the context error code is cleared.  Any options specified at
      98                 :            : /// creation remain as they were.
      99                 :            : ///
     100                 :            : /// \section Assembler
     101                 :            : ///
     102                 :            : /// The inline assembler implements each PORE/PGAS instruction as individual
     103                 :            : /// function calls.  The APIs are consistently named \c pore_\<OPCODE\>, where
     104                 :            : /// \c \<OPCODE\> is a PGAS mnemonic in upper case.  The arguments to each
     105                 :            : /// opcode appear in the same order that they appear in the source-level
     106                 :            : /// assembler, with appropriate C-language types. The supported opcode APIs
     107                 :            : /// are defined in pore_inline.h
     108                 :            : /// 
     109                 :            : /// Since the PORE instruction APIs are effectivly predicates, linear code
     110                 :            : /// sequences are easily assembled using the C-language logical OR construct.
     111                 :            : /// Any non-0 return code will immediatly break the sequence and set the
     112                 :            : /// expression value to 1.  The failure code can then be recovered from the \a
     113                 :            : /// error field of the context.  This coding technique is illustrated in the
     114                 :            : /// following example of assembling a memory-memory copy sequence.
     115                 :            : ///
     116                 :            : /// \code 
     117                 :            : ///
     118                 :            : /// PoreInlineContext ctx;
     119                 :            : /// int error;
     120                 :            : ///
     121                 :            : /// . . . // Initialize context
     122                 :            : ///
     123                 :            : /// error =
     124                 :            : ///     pore_LD(&ctx, D0, 0, A0) ||
     125                 :            : ///     pore_STD(&ctx, D0, 0, A1);
     126                 :            : ///
     127                 :            : /// if (error) <. . . Handle error based on ctx.error>
     128                 :            : ///
     129                 :            : /// \endcode
     130                 :            : ///
     131                 :            : /// The above example generates code equivalent to
     132                 :            : ///
     133                 :            : /// \code
     134                 :            : ///
     135                 :            : ///         ld      D0, 0, A0
     136                 :            : ///         std     D0, 0, A1
     137                 :            : ///
     138                 :            : /// \endcode
     139                 :            : ///
     140                 :            : /// Again, if an error were to occur during assembly, inline assembly would
     141                 :            : /// stop (and the logical OR would terminate) at the point of failure. In
     142                 :            : /// particular, the inline assembler will never allow assembled code to exceed
     143                 :            : /// the bounds of the memory area defined by the initial call of
     144                 :            : /// pore_inline_context_create() that defines the assembler memory space.
     145                 :            : ///
     146                 :            : ///
     147                 :            : /// \subsection Register Names and Other Mnemonics
     148                 :            : ///
     149                 :            : /// The header file pore_inline.h defines macros for the register mnemonics.
     150                 :            : ///
     151                 :            : /// - D0, D1 : 64-bit data registers
     152                 :            : /// - A0, A1 : 32-bit address registers
     153                 :            : /// - P0, P1 : 7-bit Pervasive chiplet id registers
     154                 :            : /// - CTR : 24-bit ounter register
     155                 :            : /// - PC : 48-bit Program Counter
     156                 :            : /// - ETR : 64-bit EXE-Trigger Register (Low-order 32 bits are writable)
     157                 :            : /// - EMR : The Error Mask Register
     158                 :            : /// - IFR : ID/Flags Register
     159                 :            : /// - SPRG0 : 32-bit Special-Purpose General Register 0
     160                 :            : ///
     161                 :            : /// Mnemonics for the condition code bits are also defined by pore_inline.h
     162                 :            : /// using the PGAS mnemonics.
     163                 :            : ///
     164                 :            : ///
     165                 :            : /// \subsection Assembling Branches
     166                 :            : ///
     167                 :            : /// Opcodes that implement relative branches require that the branch target be
     168                 :            : /// specified as a <em> location counter </em>. Once initialized, the current
     169                 :            : /// location counter is available as the \a lc field of the PoreInlineContext
     170                 :            : /// object controlling the assembly.  The \a lc field is the only field
     171                 :            : /// (besides the error code held in the \a error field) that application code
     172                 :            : /// should ever reference. The inline assembler also provides a typedef
     173                 :            : /// PoreInlineLocation to use for location counters, as well as the macro
     174                 :            : /// PORE_LOCATION() to define a location variable inline with the code flow.
     175                 :            : /// 
     176                 :            : /// \subsubsection Backward Branches
     177                 :            : ///
     178                 :            : /// Backward branches are straightforward.  For example, the memory-memory
     179                 :            : /// copy example from earlier can be converted into a loop as shown below.  The
     180                 :            : /// \a loop_target variable is initialized with the location counter of the
     181                 :            : /// first instruction of the loop.  The final instruction of the loop then
     182                 :            : /// branches back to the \a loop_target.
     183                 :            : ///
     184                 :            : /// \code
     185                 :            : ///
     186                 :            : /// PoreInlineContext ctx;
     187                 :            : /// PoreInlineLocation loop_target = 0; // See ** below the example
     188                 :            : /// int error;
     189                 :            : ///
     190                 :            : /// . . . // Initialize context
     191                 :            : ///
     192                 :            : /// error =
     193                 :            : ///     PORE_LOCATION(&ctx, loop_target) ||
     194                 :            : ///     pore_LD(&ctx, D0, 0, A0)         ||
     195                 :            : ///     pore_STD(&ctx, D0, 0, A1)            ||
     196                 :            : ///     pore_ADDS(&ctx, A0, A0, 8)           ||
     197                 :            : ///     pore_ADDS(&ctx, A1, A1, 8)           ||
     198                 :            : ///     pore_LOOP(&ctx, loop_target);
     199                 :            : ///
     200                 :            : /// if (error) <. . . Handle error based on ctx.error>
     201                 :            : ///
     202                 :            : /// \endcode
     203                 :            : ///
     204                 :            : /// The above inline assembler sequence is equivalent to the PGAS code
     205                 :            : /// sequence:
     206                 :            : ///
     207                 :            : /// \code
     208                 :            : ///
     209                 :            : /// loop_target:
     210                 :            : ///        ld      D0, 0, A0
     211                 :            : ///        std     D0, 0, A1
     212                 :            : ///        adds    A0, A0, 8
     213                 :            : ///        adds    A1, A1, 8
     214                 :            : ///        loop    loop_target
     215                 :            : ///
     216                 :            : /// \endcode
     217                 :            : ///
     218                 :            : /// ** Location counters used as loop targets may need to be initialized,
     219                 :            : /// otherwise the compiler may issue a warning that the variable "may be used
     220                 :            : /// uninitialized", although in well-written code this would never happen.
     221                 :            : ///

     223                 :            : /// \subsubsection Forward Branches
     224                 :            : ///
     225                 :            : /// Forward branches are more complex.  Since the target location counter is
     226                 :            : /// not known until the target has been assembled, the inline assembler
     227                 :            : /// provides the API pore_inline_branch_fixup() to fix up forward branches
     228                 :            : /// once the actual target is known.  This is illustrated in the simple code
     229                 :            : /// sequence below, where an instruction is conditionally skipped.
     230                 :            : ///
     231                 :            : /// \code
     232                 :            : ///
     233                 :            : /// PoreInlineContext ctx;
     234                 :            : /// PoreInlineLocation source = 0, target = 0;
     235                 :            : /// int error, rc;
     236                 :            : ///
     237                 :            : /// . . . // Initialize context
     238                 :            : ///
     239                 :            : /// error = 
     240                 :            : ///     PORE_LOCATION(&ctx, source)  ||
     241                 :            : ///     pore_BRANZ(&ctx, D0, source) ||
     242                 :            : ///     pore_ADDS(&ctx, D1, D1, 1)   ||
     243                 :            : ///     PORE_LOCATION(&ctx, target)  ||
     244                 :            : ///     pore_LD(&ctx, D0, 0, A0);
     245                 :            : ///
     246                 :            : /// if (error) <. . . Handle assembly error based on ctx->error>
     247                 :            : /// rc = pore_inline_branch_fixup(&ctx, source, target);
     248                 :            : /// if (rc) <. . . Handle branch fixup error>
     249                 :            : ///
     250                 :            : /// \endcode
     251                 :            : ///
     252                 :            : /// In the above code, the branch instruction is initially assembled as a
     253                 :            : /// branch-to-self - the recommended idiom for forward branch source
     254                 :            : /// instructions.  Once the entire sequence has been assembled,
     255                 :            : /// pore_inline_branch_fixup() reassembles the \c source instruction as a
     256                 :            : /// branch to the \c target instruction. The above instruction sequence is
     257                 :            : /// equivalent to the PGAS code below:
     258                 :            : ///
     259                 :            : /// \code
     260                 :            : ///
     261                 :            : /// source:
     262                 :            : ///         branz   D0, target
     263                 :            : ///         adds    D1, D1, 1
     264                 :            : /// target:
     265                 :            : ///         ld      D0, 0, A0
     266                 :            : ///
     267                 :            : /// \endcode
     268                 :            : ///

     270                 :            : /// \subsubsection Absolute Branches
     271                 :            : ///
     272                 :            : /// It is unlikely that a typical application of the PORE inline assembler
     273                 :            : /// would ever need to include an absolute branch, since the branch target in
     274                 :            : /// this case is a fixed absolute address that must be known at assembly
     275                 :            : /// time. However the inline assembler does provide the pore_BRAIA() API for
     276                 :            : /// this purpose.  This opcode requires a 16-bit address space constant and a
     277                 :            : /// 32-bit absoulte address (offset) within the memory space to specify the
     278                 :            : /// branch.

     281                 :            : /// \section Disassembly
     282                 :            : ///
     283                 :            : /// Inline disassembly is implemented by a single API,
     284                 :            : /// pore_inline_disassemble(). The idea is similar to assembly: A host memory
     285                 :            : /// context containing PORE code (or data) is described by a PoreInlineContext
     286                 :            : /// structure.  Each call of pore_inline_disassemble() disassembles the next
     287                 :            : /// instruction (or datum) in the context into a PoreInlineDisassembly
     288                 :            : /// structure provided by the caller.  The disassembly object contains both
     289                 :            : /// binary and string forms of the disassembled instruction (or data). The
     290                 :            : /// next call of pore_inline_disassemble() proceses the next instruction (or
     291                 :            : /// datum) and so on.
     292                 :            : ///
     293                 :            : /// \subsection Text (Code) Disassembly
     294                 :            : ///
     295                 :            : /// In the example below the inline disassembler is used to completely
     296                 :            : /// disassemble a memory area containing text (code) to \a stdout until an
     297                 :            : /// error occurs, assumed to be either due to disassembling the entire memory
     298                 :            : /// area or finding an illegal instruction.
     299                 :            : ///
     300                 :            : /// \code
     301                 :            : ///
     302                 :            : /// PoreInlineContext ctx;
     303                 :            : /// PoreInlineDisassembly dis;
     304                 :            : ///
     305                 :            : /// . . . // Initialize context
     306                 :            : ///
     307                 :            : /// while (pore_inline_disassemble(&ctx, &dis) == 0) {

     309                 :            : /// }
     310                 :            : ///
     311                 :            : /// \endcode
     312                 :            : ///
     313                 :            : /// To illustrate binary disassembly, the following example uses the
     314                 :            : /// disassembler to search for a RET statement in a block of PORE code, in
     315                 :            : /// order to extend an inline subroutine with more code.  Note that the field
     316                 :            : /// \a dis->ctx contains the context that existed at the time the instruction
     317                 :            : /// was assembled.  By copying this context back into the global context,
     318                 :            : /// inline assembly will continue by overwriting the RET with new
     319                 :            : /// instructions. If the copy had \e not been done, then newly assembled code
     320                 :            : /// would have \e followed the RET.
     321                 :            : ///
     322                 :            : /// \code
     323                 :            : ///
     324                 :            : /// PoreInlineContext ctx;

     338                 :            : /// \endcode
